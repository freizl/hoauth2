instance Show OAuth2Token where
  show OAuth2Token {..} =
    "OAuth2Token {"
      <> "access_token = ***"
      <> ", id_token = "
      <> showM idToken
      <> ", refresh_token = "
      <> showM refreshToken
      <> ", expires_in = "
      <> show expiresIn
      <> ", token_type = "
      <> show tokenType
      <> ", scope = "
      <> show scope
      <> ", raw_response = ***"
      <> "}"
    where
      showM (Just _) = "***"
      showM Nothing = "Nothing"
